核心数据结构

共享内存中的检测结果
// 单次检测结果 (共享内存，每帧更新)
typedef struct {
    uint32_t  frame_id;              // 帧序号
    uint32_t  timestamp_ms;          // 时间戳
    uint32_t  count;                 // 检测到的目标数
    DetObject objects[MAX_DET];      // 最多 64 个目标
} DetResult;

// 单个检测目标
typedef struct {
    int32_t  class_id;               // 0=person, 1=bicycle, 2=motorcycle
    float    conf;                   // 置信度 [0,1]
    float    cx, cy;                 // 底边中点 (用于区域判定)
    float    x1, y1, x2, y2;        // bbox (原图坐标)
} DetObject;

报警引擎 API

libalarm_engine.so 对外接口
// 1. 初始化，绑定检测引擎
int alarm_init(const char* shm_name);

// 2. 设置报警区域 (4 点标定, 级别 1/2/3)
int alarm_set_zone(int zone_level,           // 1/2/3
                    float pts[4][2]);         // 4 个 (x,y) 归一化坐标

// 3. 开始/停止报警检测
int alarm_start();
int alarm_stop();

// 4. 注册报警回调 (目标进入区域且停留 > 1s)
typedef void (*alarm_callback_t)(const AlarmEvent* event);
int alarm_register_callback(alarm_callback_t cb);

// 5. 销毁
void alarm_deinit();
